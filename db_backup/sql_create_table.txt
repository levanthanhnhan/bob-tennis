-- Table: public.level

-- DROP TABLE IF EXISTS public.level;

CREATE TABLE IF NOT EXISTS public.level
(
    level_id bigint NOT NULL,
    level_name character varying COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT level_pkey PRIMARY KEY (level_id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.level
    OWNER to postgres;
	


-- Table: public.role

-- DROP TABLE IF EXISTS public.role;

CREATE TABLE IF NOT EXISTS public.role
(
    role_id bigint NOT NULL,
    role_name character varying COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT role_pkey PRIMARY KEY (role_id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.role
    OWNER to postgres;
	
	
-- Table: public.members

-- DROP TABLE IF EXISTS public.members;

CREATE TABLE IF NOT EXISTS public.members
(
    member_id bigint NOT NULL,
    member_name character varying COLLATE pg_catalog."default" NOT NULL,
    birthday timestamp without time zone NOT NULL,
    phone_number character varying COLLATE pg_catalog."default" NOT NULL,
    contact character varying COLLATE pg_catalog."default" NOT NULL,
    is_active boolean NOT NULL,
    is_admin boolean NOT NULL,
    img_path character varying COLLATE pg_catalog."default" NOT NULL,
    level_id bigint,
    role_id bigint,
    CONSTRAINT members_pkey PRIMARY KEY (member_name)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.members
    OWNER to postgres;
